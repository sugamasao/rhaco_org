choice型に使用できる値の一覧を取得する

		<?php
		include("jump.php");
		
		/**
		 * @var choice $abc @{"choices":["abc","def","ghi"]}
		 */
		class Sample extends Object{
			protected $abc;
		}
		
		$obj = new Sample();
		var_dump($obj->a("abc","!!!choices!!!"));
結果
		array(3) {
		  [0]=>
		  string(3) "abc"
		  [1]=>
		  string(3) "def"
		  [2]=>
		  string(3) "ghi"
		}
