Httpクラスを利用する

GETで取得する

		<?php
		include 'http.php';
		$http = new Http('Mozilla/5.0');
		$http->header('Referer','http://rhaco.org');
		$http->vars('query','Http');
		$http->do_get('http://rhaco.org/repository/3/lib');
		
		print($http->body());

POSTで取得する

		<?php
		include 'http.php';
		$http = new Http('Mozilla/5.0');
		$http->header('Referer','http://rhaco.org');
		$http->vars('query','Http');
		$http->do_post('http://rhaco.org/repository/3/lib');
		
		print($http->body());
